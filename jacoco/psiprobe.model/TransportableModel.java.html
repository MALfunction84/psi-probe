package psiprobe.model;

import java.util.Collections;
import java.util.HashMap;
import java.util.Map;

/**
 * A wrapper class to assist marshalling of ModelAndView.getModel() map to XML representation.
 */
public class TransportableModel {

  /** The items. */
  private Map<String, Object> items = new HashMap<>();

  /**
   * Gets the items.
   *
   * @return the items
   */
  public Map<String, Object> getItems() {
    return items == null ? Collections.emptyMap() : new HashMap<>(items);
  }

  /**
   * Sets the items.
   *
   * @param items the items
   */
  public void setItems(Map<String, Object> items) {
    this.items = items == null ? Collections.emptyMap() : new HashMap<>(items);
  }

  /**
   * Put all.
   *
   * @param map the map
   */
  public void putAll(Map<String, Object> map) {
    if (map != null) {
      items.putAll(new HashMap<>(map));
    }
  }

}
